Following query uses the function to do various offsets of time:

| source empty
| set
	now =timeoffset() # returns epoch milliseconds of present time 
	oneYear = timeoffset(null, 1) #takes current time and offsets it by +1 year
	oneMonth = timeoffset($latest$, 0, -1) #takes latest time and and offsets it by -1 month
	oneDay = timeoffset($earliest$, 0, 0, -1) #takes earliest time and and offsets it by -1 day
	oneHour = timeoffset($earliest$, 0, 0, 0, -1) #takes earliest time and and offsets it by -1 hour
	oneMinute = timeoffset($earliest$, 0, 0, 0, 0, -1) #takes earliest time and and offsets it by -1 day
	oneSecond = timeoffset($earliest$, 0, 0, 0, 0, 0, -1) #takes earliest time and and offsets it by -1 second
	fiftyMilli = timeoffset($earliest$, 0, 0, 0, 0, 0, 0, -50) #takes earliest time and and offsets it by -1 millisecond
| formatfield
	# just some formatters
	[now] = ['timestamp', "yyyy-MM-dd HH:mm:ss"]
	[oneYear, oneMonth, oneDay] = ['date']
	[oneHour, oneMinute, oneSecond, fiftyMilli] = ['timestamp', "HH:mm:ss.SSS"]
